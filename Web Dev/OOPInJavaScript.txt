// Objects in JavaScript


const circle = {

  radius: 2,

  location: {
    x: 2,
    y: 2
  },

  draw: function() {

    console.log('A Drawing of the circle');

  }

};

circle.draw();


// The FOR...IN Loop in JavaScript

for (var key in circle) {

  console.log(key); 
  
  // to show what is inside the circle object

  console.log(key, circle[key]); 
  
  // to show the exact value of the variables in the cicle object

  if (typeof circle[key] !== 'function')
      console.log(key, circle[key]);

}

const keys = Object.keys(circle);

console.log(keys);

if ('radius' in circle)
    console.log('The circle has a radius.');
